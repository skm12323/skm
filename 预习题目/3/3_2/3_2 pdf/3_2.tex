\[
f(x)=A\sin(x+B)+C
\]
\[
f'(x)=A\cos(x+B)
\]
\[
\frac{f(x+\delta)-f(x-\delta)}{\tau} = A\cdot\frac{\sin(x+\delta+B)-\sin(x-\delta+B)}{\tau}= \frac{2A}{\tau}\cdot \sin(\delta) \cos(x+B)
\]
若:
\[
f'(x)=\frac{f(x+\delta)-f(x-\delta)}{\tau}
\]
则有:
\[
\tau = 2\sin(\delta)
\]
